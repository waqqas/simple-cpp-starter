cmake_minimum_required(VERSION 3.10)

# Set C++ standard

# Enable all warnings and treat them as errors
if(MSVC)
  add_compile_options(/W4 /WX)
else()
  add_compile_options(-Wall -Wextra -pedantic -Werror)
endif()
